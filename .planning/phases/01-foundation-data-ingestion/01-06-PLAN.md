```python
from celery import Celery
from src.core.config import settings
from src.core.logging import get_logger

logger = get_logger(__name__)

celery_app = Celery(
    "vbrain",
    broker=settings.redis_url,
    backend=settings.redis_url,
)

@celery_app.task(bind=True, autoretry_for=(Exception,), max_retries=3)
def ingest_telegram(self, source_id: str, json_path: str, voice_dir: str):
    """Full Telegram ingestion pipeline."""
    try:
        # Stage 1: Parse (ING-02)
        self.update_state(state="PROGRESS", meta={"stage": "parsing", "progress": 10})
        messages = parse_telegram_export(json_path)
        logger.info(f"Parsed {len(messages)} messages from Telegram export")

        # Stage 2: Transcribe voice messages (ING-03)
        self.update_state(state="PROGRESS", meta={"stage": "transcribing", "progress": 20})
        messages = transcribe_voice_messages(messages, voice_dir)

        # Stage 3: PII Anonymization -- BEFORE any processing (D-07, ING-01)
        self.update_state(state="PROGRESS", meta={"stage": "anonymizing", "progress": 35})
        analyzer = create_analyzer()
        anonymizer = create_anonymizer()
        for msg in messages:
            msg.text = anonymize_text(msg.text, analyzer, anonymizer)

        # Stage 4: Noise filtering (ING-04, D-09/D-10)
        self.update_state(state="PROGRESS", meta={"stage": "filtering", "progress": 50})
        messages = filter_messages(messages)

        # Stage 5: Chronological grouping + chunking (D-12, D-15)
        self.update_state(state="PROGRESS", meta={"stage": "chunking", "progress": 65})
        chunks = chunk_telegram_messages(messages)

        # Stage 6: Embedding + Indexing (ING-06)
        self.update_state(state="PROGRESS", meta={"stage": "indexing", "progress": 80})
        embedder = Embedder()
        store = QdrantStore(QdrantClient(host=settings.qdrant_host, port=settings.qdrant_port))
        store.ensure_collection()

        # Add source_id to metadata
        for chunk in chunks:
            chunk["metadata"]["source_id"] = source_id

        # Generate embeddings and upsert
        texts = [c["text"] for c in chunks]
        dense_vectors, sparse_vectors = embedder.embed_batch(texts)

        for i, chunk in enumerate(chunks):
            chunk["dense_vector"] = dense_vectors[i].tolist()
            chunk["sparse_vector"] = sparse_vectors[i]
            chunk["id"] = f"telegram:{source_id}:{i}"

        count = store.upsert_chunks(chunks)
        logger.info(f"Indexed {count} chunks from Telegram source {source_id}")

        return {
            "status": "completed",
            "messages_processed": len(messages),
            "chunks_indexed": count,
        }
    except Exception as exc:
        logger.error(f"Ingestion failed for source {source_id}: {exc}")
        self.update_state(state="FAILURE", meta={"error": str(exc)})
        raise

@celery_app.task(bind=True, autoretry_for=(Exception,), max_retries=3)
def ingest_pdf(self, source_id: str, file_path: str):
    """Full PDF ingestion pipeline."""
    try:
        # Stage 1: Parse PDF (ING-05)
        self.update_state(state="PROGRESS", meta={"stage": "parsing", "progress": 15})
        markdown = extract_pdf_text(file_path)

        # Stage 2: PII Anonymization (D-07, ING-01)
        self.update_state(state="PROGRESS", meta={"stage": "anonymizing", "progress": 30})
        analyzer = create_analyzer()
        anonymizer = create_anonymizer()
        markdown = anonymize_text(markdown, analyzer, anonymizer)

        # Stage 3: Chunking (D-12, D-14)
        self.update_state(state="PROGRESS", meta={"stage": "chunking", "progress": 50})
        text_chunks = chunk_text(markdown)
        chunks = []
        for idx, tc in enumerate(text_chunks):
            chunks.append({
                "text": tc,
                "metadata": {
                    "source_type": "pdf",
                    "source_id": source_id,
                    "chunk_index": idx,
                },
            })

        # Stage 4: Embedding + Indexing
        self.update_state(state="PROGRESS", meta={"stage": "indexing", "progress": 75})
        embedder = Embedder()
        store = QdrantStore(QdrantClient(host=settings.qdrant_host, port=settings.qdrant_port))
        store.ensure_collection()

        texts = [c["text"] for c in chunks]
        dense_vectors, sparse_vectors = embedder.embed_batch(texts)

        for i, chunk in enumerate(chunks):
            chunk["dense_vector"] = dense_vectors[i].tolist()
            chunk["sparse_vector"] = sparse_vectors[i]
            chunk["id"] = f"pdf:{source_id}:{i}"

        count = store.upsert_chunks(chunks)
        logger.info(f"Indexed {count} chunks from PDF source {source_id}")

        return {"status": "completed", "chunks_indexed": count}
    except Exception as exc:
        logger.error(f"PDF ingestion failed for source {source_id}: {exc}")
        self.update_state(state="FAILURE", meta={"error": str(exc)})
        raise
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| User -> REST API | File uploads (untrusted input crosses here) |
| REST API -> Celery | Task dispatch with file paths |
| Celery -> External APIs | Groq Whisper (voice transcription) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-17 | Tampering | File upload endpoints | mitigate | Validate file extension (.json, .pdf, .ogg). Validate magic bytes for PDF (%PDF-). Reject invalid types with 400. Limit size to settings.max_file_size_mb (default 100MB). Return 413 for oversized |
| T-01-18 | Tampering | File path traversal | mitigate | Use uuid4() for saved filenames. Never use user-supplied filename for filesystem operations. Save to isolated upload_dir |
| T-01-19 | Denial of Service | Concurrent uploads | accept | No concurrent limit in Phase 1. Single-tenant admin usage. Future: add rate limiting |
| T-01-20 | Information Disclosure | API error messages | mitigate | Generic error messages to client. Detailed errors only in server logs (structured logging) |
| T-01-21 | Tampering | job_id in status endpoint | accept | UUID job_ids are non-guessable. No authorization in Phase 1 (added in Phase 3/4) |
</threat_model>
