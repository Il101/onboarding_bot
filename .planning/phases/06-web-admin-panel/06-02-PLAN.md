Add PDF upload endpoint:

```python
@router.post("/sources/pdf", response_class=HTMLResponse)
async def admin_upload_pdf(
    request: Request,
    file: UploadFile = File(...),
    db: Session = Depends(get_db_session),
):
    filename = (file.filename or "").lower()
    if not filename.endswith(".pdf"):
        return templates.TemplateResponse(
            "sources/_upload_status.html",
            {"request": request, "success": False, "message": "Неверный тип файла. Ожидается .pdf"},
        )

    try:
        content = await file.read()
        _validate_size(content)
        if not content.startswith(b"%PDF-"):
            return templates.TemplateResponse(
                "sources/_upload_status.html",
                {"request": request, "success": False, "message": "Неверное содержимое PDF файла"},
            )

        base = _ensure_upload_dirs()
        source_id = str(uuid.uuid4())
        safe_name = f"{source_id}.pdf"
        file_path = base / safe_name
        file_path.write_bytes(content)

        source = Source(
            id=source_id,
            type=SourceType.PDF,
            filename=file.filename or "unknown.pdf",
            file_path=str(file_path),
            status=IngestStatus.PENDING,
        )
        db.add(source)
        db.commit()

        task = ingest_pdf.delay(source_id, str(file_path))
        job = IngestJob(
            id=str(uuid.uuid4()),
            source_id=source_id,
            celery_task_id=task.id,
            status="PENDING",
            progress=0,
        )
        db.add(job)
        db.commit()

        return templates.TemplateResponse(
            "sources/_upload_status.html",
            {"request": request, "success": True, "message": f"PDF загружен. Job ID: {task.id}", "job_id": task.id},
        )
    except HTTPException:
        raise
    except Exception as e:
        return templates.TemplateResponse(
            "sources/_upload_status.html",
            {"request": request, "success": False, "message": f"Ошибка загрузки: {str(e)}"},
        )
```

```python
@router.post("/sources/telegram", response_class=HTMLResponse)
async def admin_upload_telegram(
    request: Request,
    json_file: UploadFile = File(...),
    voice_files: list[UploadFile] = File(default=[]),
    db: Session = Depends(get_db_session),
):
    if not (json_file.filename or "").lower().endswith(".json"):
        return templates.TemplateResponse(
            "sources/_upload_status.html",
            {"request": request, "success": False, "message": "Неверный тип файла. Ожидается .json"},
        )

    try:
        base = _ensure_upload_dirs()
        source_id = str(uuid.uuid4())
        json_name = f"{source_id}.json"
        json_path = base / json_name
        json_content = await json_file.read()
        _validate_size(json_content)
        _validate_json_content(json_content)
        json_path.write_bytes(json_content)

        voice_dir = base / f"{source_id}_voices"
        voice_dir.mkdir(parents=True, exist_ok=True)
        for vf in voice_files:
            if not (vf.filename or "").lower().endswith(".ogg"):
                return templates.TemplateResponse(
                    "sources/_upload_status.html",
                    {"request": request, "success": False, "message": "Неверный тип голосового файла. Ожидается .ogg"},
                )
            content = await vf.read()
            _validate_size(content)
            _validate_ogg_content(content)
            safe_name = f"{uuid.uuid4()}.ogg"
            (voice_dir / safe_name).write_bytes(content)

        source = Source(
            id=source_id,
            type=SourceType.TELEGRAM,
            filename=json_file.filename or "result.json",
            file_path=str(json_path),
            status=IngestStatus.PENDING,
        )
        db.add(source)
        db.commit()

        task = ingest_telegram.delay(source_id, str(json_path), str(voice_dir))
        job = IngestJob(
            id=str(uuid.uuid4()),
            source_id=source_id,
            celery_task_id=task.id,
            status="PENDING",
            progress=0,
        )
        db.add(job)
        db.commit()

        return templates.TemplateResponse(
            "sources/_upload_status.html",
            {"request": request, "success": True, "message": f"Telegram логи загружены. Job ID: {task.id}", "job_id": task.id},
        )
    except HTTPException:
        raise
    except Exception as e:
        return templates.TemplateResponse(
            "sources/_upload_status.html",
            {"request": request, "success": False, "message": f"Ошибка загрузки: {str(e)}"},
        )
```

```python
@router.delete("/sources/{source_id}", response_class=HTMLResponse)
async def delete_source(source_id: str, db: Session = Depends(get_db_session)):
    source = db.query(Source).filter(Source.id == source_id).first()
    if not source:
        return HTMLResponse(content='<div class="bg-red-50 text-red-700 p-3 rounded">Источник не найден</div>', status_code=404)
    db.delete(source)
    db.commit()
    return HTMLResponse(content='<div class="bg-green-50 text-green-700 p-3 rounded">Источник удалён</div>')
```
